–- 3) Фільтр по employee_id --
SELECT * FROM ( 
    SELECT od.id AS order_detail_id, o.employee_id, cat.name AS category_name, od.quantity
    FROM order_details od 
    INNER JOIN orders o ON od.order_id = o.id 
    INNER JOIN products p ON od.product_id = p.id 
    INNER JOIN categories cat ON p.category_id = cat.id) t 
WHERE employee_id > 3 AND employee_id <= 10;

–- 4) Групування --
SELECT
    cat.name AS category_name,
    COUNT(*) AS row_count,
    AVG(od.quantity) AS avg_quantity
FROM order_details od
INNER JOIN products p     ON od.product_id = p.id
INNER JOIN categories cat ON p.category_id = cat.id
GROUP BY cat.name;

-- 5) Отфільтровані рядки, де середня кількість товару більша за 21 --
SELECT
    cat.name AS category_name,
    COUNT(*) AS row_count,
    AVG(od.quantity) AS avg_quantity
FROM order_details od
INNER JOIN products p     ON od.product_id = p.id
INNER JOIN categories cat ON p.category_id = cat.id
GROUP BY cat.name
HAVING AVG(od.quantity) > 21;

-- 6) Відсортовані рядки за спаданням кількості рядків --
SELECT
    cat.name AS category_name,
    COUNT(*) AS row_count,
    AVG(od.quantity) AS avg_quantity
FROM order_details od
INNER JOIN products p     ON od.product_id = p.id
INNER JOIN categories cat ON p.category_id = cat.id
GROUP BY cat.name
HAVING AVG(od.quantity) > 21
ORDER BY row_count DESC;

-- 7) Чотири рядки з пропущеним першим рядком --
SELECT
    cat.name AS category_name,
    COUNT(*) AS row_count,
    AVG(od.quantity) AS avg_quantity
FROM order_details od
INNER JOIN products p     ON od.product_id = p.id
INNER JOIN categories cat ON p.category_id = cat.id
GROUP BY cat.name
HAVING AVG(od.quantity) > 21
ORDER BY row_count DESC
LIMIT 4 OFFSET 1;
